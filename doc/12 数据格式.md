## 数据格式

主要介绍的是JSON、XML、MSGPack

### 1 JSON

```go
1 json是完全独立于语言的文本格式，是k-v的形式  name:zs
2 应用场景：前后端交互，系统间数据交互
3 json使用go语言内置的encoding/json 标准库
4 编码json使用json.Marshal()函数可以对一组数据进行JSON格式的编码
func Marshal(v interface{}) ([]byte, error)
```

> 通过结构体生成json

```go
package main

import (
	"encoding/json"
	"fmt"
)

type Person struct {
	Name string //如果写成name 就是私有的 main函数中就是空 
	Nobby string
}

func main() {
	p := Person{"zhangsan", "dbj"}
	// 返回的是byte的切片
	b, err := json.Marshal(p)
	if err != nil{
		fmt.Println("json err", err)
	}
	fmt.Println(string(b)) // string 这个方法好像有问题
	// 格式化输出
	r, e := json.MarshalIndent(p, "", "")
	if err != nil{
		fmt.Println("json error", e)
	}
	fmt.Println(string(r))
}
```

> 通过map 生成json

```go
func main() {
	// 创建map
	mmp := make(map[string]interface{})
	mmp["name"] = "zs"
	mmp["age"] = 18
	// map --json
	mjson, err := json.Marshal(mmp)
	if err != nil{
		fmt.Println("json err", err)
	}
	fmt.Println(string(mjson))
}
```

```go
)

type Person struct {
	Age int `json:"age"`
```

> 将json 解析到interface, 自动转到map

```go
package main

import (
   "encoding/json"
   "fmt"
)

func main() {
   // 假数据
   // int和float64都当float64
   b := []byte(`{"age":1.3,"name":"Mr.Sun","niubility":true}`)

   // 声明接口
   var i interface{}
   err := json.Unmarshal(b, &i)
   if err != nil {
      fmt.Println(err)
   }
   // 自动转到map
   fmt.Println(i)
   // 可以判断类型
   m := i.(map[string]interface{})
   for k, v := range m {
      switch vv := v.(type) {
      case float64:
         fmt.Println(k, "是float64类型", vv)
      case string:
         fmt.Println(k, "是string类型", vv)
      default:
         fmt.Println("其他")
      }
   }
}
```

### 2 XML

```html
是可扩展标记语言，包含声明、根标签、子元素和属性
应用场景：配置文件以及webService
```

> 读取xml

```xml
<?xml version="1.0" encoding="UTF-8" ?>
<servers version="1">
    <server>
        <serverName>Shanghai_VPN</serverName>
        <serverIP>127.0.0.1</serverIP>
    </server>
    <server>
        <serverName>Beijing_VPN</serverName>
        <serverIP>127.0.0.2</serverIP>
    </server>
</servers>
```

```go
package main

import (
	"encoding/xml"
	"fmt"
	"io/ioutil"
)

// 单个的server
type Server struct {
	ServerName string `xml:"serverName"`
	ServerIP   string `xml:"serverIP"`
}
// 整体的
type Servers struct {
	Name    xml.Name `xml:servers`
	Version string   `xml:"version"`
	Servers []Server `xml:"server"` // 多个单个的Server
}

func main() {
	data, err := ioutil.ReadFile("my.xml")
	if err != nil{
		fmt.Println(err)
		return
	}
	var servers Servers
	err = xml.Unmarshal(data, &servers)
	if err != nil{
		fmt.Println(err)
		return
	}
	fmt.Printf("%#v\n", servers)
	fmt.Println(servers.Name)
}
```

### MSGPack

```html
MSGPack是二进制的json，性能更快，更加节省空间
需要安装第三方的包：go get -u -v github.com/vmihailenco/msgpack
```

```go
package main

import (
	"fmt"
	"io/ioutil"
	"math/rand"
	"github.com/vmihailenco/msgpack"
)

type Person struct {
	Name string
	Age  int
	Sex  string
}

// 二进制写出
func writeJson(filename string) (err error) {
	var persons []*Person
	// 假数据
	for i := 0; i < 10; i++ {
		p := &Person{
			Name: fmt.Sprintf("name%d", i),
			Age:  rand.Intn(100),
			Sex:  "male",
		}
		persons = append(persons, p)
	}
	// 二进制json序列化
	data, err := msgpack.Marshal(persons)
	if err != nil {
		fmt.Println(err)
		return
	}

	err2 := ioutil.WriteFile(filename, data, 0666)
	if err2 != nil {
		fmt.Println(err2)
		return
	}
	return
}

// 二进制读文件
func readJson(filename string) (err error) {
	var persons []*Person
	// 读文件
	data, err := ioutil.ReadFile(filename)
	if err != nil {
		fmt.Println(err)
		return
	}
	// 反序列化
	err2 := msgpack.Unmarshal(data, &persons) // 取地址，
	if err2 != nil {
		fmt.Println(err2)
		return
	}
	// 遍历
	for _, v := range persons{
		fmt.Printf("%#v\n", v)
	}
	return
}

func main() {
	// 写
	//err := writeJson("aaa.dat")
	//if err != nil {
	//	fmt.Println(err)
	//	return
	//}
	// 读
	err := readJson("aaa.dat")
	if err != nil {
		fmt.Println(err)
		return
	}
}
```
