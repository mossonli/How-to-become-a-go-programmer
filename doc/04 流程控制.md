> if、if-else、if-else if-else

```go
func main(){
	a := 100
	if a < 20 {
		fmt.Println("a小于20")
	}else {
		fmt.Println("a大于20")
	}

```

```go
func main() {

  }
}
```

## 4.2 选择语句

> switch 语句

```go
// 表达式 switch
func main() {
  grade := "A"
  marks := 80
  
  switch marks {
    case 90:
    	grade = "A"
    case 80:
    	grade = "B"
    case 70:
    	grade = "C"
  }
}
// 类型switch
var x interface {}
func main() {
	x = 1 //修改x的值，查看返回结果的变化
  switch i := x.(type) {
    case nil:
    	fmt.Println("这里是nil, x的类型是%T"， i)
  }
}
```

> select语句

```go
这种选择语句用于配合通道channel)的读写操作，用于多个channel的并发读写操作。
switch是按顺序从上到下依次执行的，而 select是随机选择一个case来判断， 直到匹配其中的一个 case
func main() {
  a := make(chan int, 1024)
  b := make(chan int, 1024)
  for i := 0; i < 10; i++ {
    fmt.Printf("第%d次"， i)
    a <-1
    b <-1
    select {
      case < -a :
      	fmt.Println("from a")
      case < -b :
      	fmt.Println("from b")
    }
  }
}
```

## 4.4 延迟语句

```go
// defer 用于延迟调用指定函数，defer 关键字只能出现在函数内部！
func mian() {
  defer fmt.Println("world")
  fmt.Println("hello")
}
/*
hello
world
*/
// 1 只有当defer语句全部执行，defer所在的函数才算真正结束
// 2 当函数中有defer语句时，需要等待所有的defer语句执行完毕，才会执行return
```

## 4.5 标签

> break

```go
// 结束当前的循环
func main(){
	LOOP1 :
		for {
			x := 1
			switch {
			case x > 0:
				fmt.Println("A")
				break LOOP1
			case x == 1:
				fmt.Println("B")
			default:
				fmt.Println("C")
			}
		}
}
// break LOOP1 表示跳出 LOOP1代码块之外
```

> continue

```go
// continue 用于跳转到指定的代码块继续执行任务，continue仅能用于for循环
func main(){
	LOOP1 :
		for i := 0; i < 5; i++ {
			switch {
			case i > 0:
				fmt.Println("A")
				continue LOOP1
			case i == 1:
				fmt.Println("B")
			default:
				fmt.Println("C")
			}
			fmt.Printf("i is %d\n", i)
		}
}
/*
i = 0的时候 执行default 打印C
执行 i is 0
进入到下一次循环 i=1
*/
```
